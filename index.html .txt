<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>نظام التحليل المالي | هدايه</title>
    <!-- Tailwind CSS for Styling -->
    <script src="https://cdn.tailwindcss.com"></script>
    <!-- Lucide Icons -->
    <script src="https://unpkg.com/lucide@latest"></script>
    <style>
        @import url('https://fonts.googleapis.com/css2?family=Noto+Sans+Arabic:wght@300;400;700;900&display=swap');
        body { font-family: 'Noto Sans Arabic', sans-serif; }
        .text-glow { text-shadow: 0 0 10px rgba(52, 211, 153, 0.5); }
        input::-webkit-outer-spin-button, input::-webkit-inner-spin-button { -webkit-appearance: none; margin: 0; }
    </style>
</head>
<body class="bg-slate-50 text-slate-900">
    <div id="app">
        <!-- سيتم بناء الواجهة بواسطة JavaScript -->
    </div>

    <script>
        // الحالة العامة للنظام (State)
        const state = {
            isAuthenticated: false,
            activeTab: 'welcome',
            financialData: {
                assets: { current: 0, fixed: 0, receivables: 0 },
                liabilities: { current: 0, longTerm: 0 },
                equity: { capital: 0, retainedEarnings: 0 },
                revenue: 0,
                cogs: 0,
                expenses: 0,
                inventory: 0,
                cashFlow: { operating: 0, investing: 0, financing: 0 }
            }
        };

        // دالة التحديث الرئيسية (Re-render)
        function updateUI() {
            const app = document.getElementById('app');
            
            if (!state.isAuthenticated) {
                app.innerHTML = renderLogin();
            } else {
                app.innerHTML = `
                    ${renderHeader()}
                    <main class="pt-24 pb-12 px-4 md:px-8 max-w-7xl mx-auto">
                        ${renderContent()}
                    </main>
                    ${renderFooter()}
                `;
            }
            // إعادة تفعيل الأيقونات
            lucide.createIcons();
        }

        // --- شاشة تسجيل الدخول ---
        function renderLogin() {
            return `
                <div class="min-h-screen flex items-center justify-center p-6 bg-slate-100">
                    <div class="w-full max-w-md bg-white rounded-[2.5rem] shadow-2xl overflow-hidden border border-slate-200">
                        <div class="bg-indigo-900 p-8 text-center text-white">
                            <div class="inline-flex p-4 bg-white/10 rounded-full mb-4">
                                <i data-lucide="lock" class="text-emerald-400 w-8 h-8"></i>
                            </div>
                            <h2 class="text-2xl font-bold">نظام التحليل المالي</h2>
                            <p class="text-indigo-200 text-sm mt-2">أدخل الرمز الخاص ب هدايه</p>
                        </div>
                        <div class="p-8 space-y-6">
                            <input type="password" id="passInput" placeholder="••••" 
                                class="w-full text-center text-2xl tracking-[1em] bg-slate-50 border-2 border-transparent focus:border-indigo-200 rounded-2xl py-4 outline-none transition-all">
                            <button onclick="handleAuth()" class="w-full bg-indigo-900 hover:bg-indigo-800 text-white font-black py-4 rounded-2xl transition-all flex items-center justify-center gap-2">
                                فتح النظام <i data-lucide="unlock" class="w-5 h-5"></i>
                            </button>
                        </div>
                    </div>
                </div>
            `;
        }

        function handleAuth() {
            const val = document.getElementById('passInput').value;
            if (val === '2003') {
                state.isAuthenticated = true;
                updateUI();
            } else {
                alert('الرمز غير صحيح');
            }
        }

        // --- مكونات النظام الداخلي ---
        function renderHeader() {
            return `
                <header class="fixed top-0 left-0 right-0 bg-white/80 backdrop-blur-md border-b border-slate-200 z-40 px-6 py-4 flex justify-between items-center">
                    <div class="flex items-center gap-3">
                        <div class="bg-indigo-900 p-2 rounded-xl text-white"><i data-lucide="shield-check"></i></div>
                        <h1 class="text-lg font-bold text-indigo-900">نظام التحليل الاستراتيجي</h1>
                    </div>
                    <div class="flex gap-4">
                        <button onclick="state.activeTab='input'; updateUI();" class="text-sm font-bold ${state.activeTab === 'input' ? 'text-indigo-600' : 'text-slate-400'}">الإدخال</button>
                        <button onclick="state.activeTab='statements'; updateUI();" class="text-sm font-bold ${state.activeTab === 'statements' ? 'text-indigo-600' : 'text-slate-400'}">القوائم</button>
                        <button onclick="state.activeTab='ratios'; updateUI();" class="text-sm font-bold ${state.activeTab === 'ratios' ? 'text-indigo-600' : 'text-slate-400'}">النسب</button>
                    </div>
                </header>
            `;
        }

        function renderContent() {
            if (state.activeTab === 'welcome') return `
                <div class="bg-gradient-to-br from-slate-900 to-indigo-900 rounded-[2.5rem] p-12 text-white shadow-2xl">
                    <h2 class="text-4xl font-light mb-6">مرحباً بكِ في عالم <span class="font-bold text-emerald-400">الوفرة والسيادة</span></h2>
                    <p class="text-indigo-100 opacity-80 mb-8 max-w-xl">أداة احترافية صُممت لتعكس ذكاءكِ المحاسبي وقدرتكِ الإدارية الرصينة.</p>
                    <button onclick="state.activeTab='input'; updateUI();" class="bg-emerald-500 px-8 py-4 rounded-2xl font-bold flex items-center gap-2">ابدأ الآن <i data-lucide="chevron-right"></i></button>
                </div>
            `;
            
            if (state.activeTab === 'input') return `
                <div class="grid grid-cols-1 lg:grid-cols-2 gap-8">
                    <div class="bg-white p-8 rounded-[2rem] border border-slate-100 shadow-sm">
                        <h3 class="text-lg font-bold mb-6 flex items-center gap-2"><i data-lucide="wallet"></i> الميزانية العمومية</h3>
                        <div class="grid grid-cols-1 md:grid-cols-2 gap-4">
                            <input type="number" onchange="updateData('assets','current',this.value)" placeholder="الأصول المتداولة" class="p-4 bg-slate-50 rounded-xl outline-none border-2 border-transparent focus:border-indigo-100">
                            <input type="number" onchange="updateData('assets','fixed',this.value)" placeholder="الأصول الثابتة" class="p-4 bg-slate-50 rounded-xl outline-none">
                            <input type="number" onchange="updateData('liabilities','current',this.value)" placeholder="التزامات متداولة" class="p-4 bg-slate-50 rounded-xl outline-none">
                            <input type="number" onchange="updateData('equity','capital',this.value)" placeholder="رأس المال" class="p-4 bg-slate-50 rounded-xl outline-none">
                        </div>
                    </div>
                    <div class="bg-white p-8 rounded-[2rem] border border-slate-100 shadow-sm">
                        <h3 class="text-lg font-bold mb-6 flex items-center gap-2"><i data-lucide="trending-up"></i> قائمة الدخل</h3>
                        <div class="grid grid-cols-1 gap-4">
                            <input type="number" onchange="updateData(null,'revenue',this.value)" placeholder="إجمالي الإيرادات" class="p-4 bg-slate-50 rounded-xl outline-none">
                            <input type="number" onchange="updateData(null,'cogs',this.value)" placeholder="تكلفة المبيعات" class="p-4 bg-slate-50 rounded-xl outline-none">
                            <input type="number" onchange="updateData(null,'expenses',this.value)" placeholder="المصاريف التشغيلية" class="p-4 bg-slate-50 rounded-xl outline-none">
                        </div>
                    </div>
                </div>
            `;

            if (state.activeTab === 'statements') {
                const totalAssets = state.financialData.assets.current + state.financialData.assets.fixed;
                const netIncome = state.financialData.revenue - state.financialData.cogs - state.financialData.expenses;
                return `
                    <div class="grid grid-cols-1 md:grid-cols-3 gap-6">
                        <div class="bg-white p-6 rounded-[2rem] shadow-sm border border-slate-100">
                            <h4 class="font-bold text-indigo-900 border-b pb-2 mb-4">قائمة الدخل</h4>
                            <div class="flex justify-between mb-2"><span>المبيعات</span><span class="font-bold">${state.financialData.revenue}</span></div>
                            <div class="flex justify-between mb-2 text-rose-500"><span>التكاليف (-)</span><span class="font-bold">${state.financialData.cogs + state.financialData.expenses}</span></div>
                            <div class="bg-indigo-900 text-white p-4 rounded-xl flex justify-between mt-4"><span>صافي الربح</span><span class="font-bold">${netIncome}</span></div>
                        </div>
                        <div class="bg-white p-6 rounded-[2rem] shadow-sm border border-slate-100">
                            <h4 class="font-bold text-indigo-900 border-b pb-2 mb-4">المركز المالي</h4>
                            <div class="flex justify-between mb-2"><span>إجمالي الأصول</span><span class="font-bold">${totalAssets}</span></div>
                            <div class="flex justify-between mb-2"><span>الالتزامات</span><span class="font-bold">${state.financialData.liabilities.current}</span></div>
                            <div class="bg-emerald-500 text-white p-4 rounded-xl flex justify-between mt-4"><span>حقوق الملكية</span><span class="font-bold">${state.financialData.equity.capital}</span></div>
                        </div>
                         <div class="bg-white p-6 rounded-[2rem] shadow-sm border border-slate-100">
                            <h4 class="font-bold text-indigo-900 border-b pb-2 mb-4">حركة النقد</h4>
                            <div class="flex justify-between mb-2"><span>تدفق تشغيلي</span><span class="font-bold">${state.financialData.cashFlow.operating}</span></div>
                            <div class="bg-slate-900 text-white p-4 rounded-xl flex justify-between mt-4"><span>صافي النقد</span><span class="font-bold">${state.financialData.cashFlow.operating}</span></div>
                        </div>
                    </div>
                `;
            }

            return `<p class="text-center text-slate-400">لوحة النسب المالية قيد المعالجة...</p>`;
        }

        function renderFooter() {
            return `<footer class="mt-20 text-center text-slate-400 text-xs">حقوق الملكية محفوظة لهدايه © 2026</footer>`;
        }

        function updateData(cat, field, val) {
            const num = parseFloat(val) || 0;
            if (cat) state.financialData[cat][field] = num;
            else state.financialData[field] = num;
            // لا نحدث الواجهة هنا مباشرة لمنع فقدان التركيز (Focus) أثناء الإدخال
        }

        // تشغيل النظام
        window.onload = updateUI;
    </script>
</body>
</html>
